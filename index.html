React Routing

Create react project.

npx create-react-app router-tutorial

Then go to the project folder.

cd  router-tutorial

And install react-router-dom.

npm install react-router-dom

Run the app.

npm start

src/App.js

import React from 'react';
import { BrowserRouter as Router, Switch, Route, Link } from 'react-router-dom';
const links = [
  ["/", "Home", Home],
  ["/about", "About", About],
  ["/project", "Project", Project]
];
export default function App() {
  return (
    <Router>
      <Nav />
      <hr />
      <Switch>
        {links.map(link => <Route key={link[1]}
          path={link[0]} exact component={link[2]} />)}
        <Route path="/Project/:id123" component={Projects} />
        <Route component={NoMatch} />
      </Switch>
    </Router>
  );
}

The rest of component in the same file.

function Nav() {
  return (
    <nav>
      <ul>
        {links.map(link =>
          <Link key={link[1]} to={link[0]}>
            <li key={link[1]}>{link[1]}</li>
          </Link>)}
      </ul>
    </nav>
  )
}
function Projects(match) { 
  return (
    <div>Project: "{match.match.params.id123}"</div>
  )
}
function Home() { return (<div>Home Page</div>) }
function About() { return (<div>About Page</div>) }
function NoMatch() { return (<div>404 Page</div>) }
function Project() {
  return (<div>
    {links.map(link => <Link key={link[1]} to={'project/' + link[1]}>
      <p key={link[1]}>{link[1]}</p>
    </Link>)}
  </div>)
}
